def have_digits(s: str) -> bool:
    """Checks if a string has digits in it

    Args:
        s: A string

    Returns:
        A boolean telling whether the string given has any digits (numbers) in it.
    """

    out = 0

    # loop through the string
    for c in s:
        # check if the character is a digit
        if c.isdigit():
            out = 1
            break

    return out
